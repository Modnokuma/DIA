  1. Escriba un programa que, dado un entero introducido por teclado, calcule todos sus divisores enteros, mostrándolos por pantalla.

    Código fuente 
    // Divisores (c) 2021 Baltasar MIT License <baltasarq@uvigo.gal>


namespace Divisores {
    using System;


    class Program {
        static void Main(string[] args)
        {
            Console.WriteLine( "Divisores\n" );

            Console.Write("Introduzca entero: ");
            int num = Convert.ToInt32( Console.ReadLine() );

            Console.WriteLine( "1" );
            for(int i = 2; i < num; ++i) {
                if ( num % i == 0 ) {
                    Console.WriteLine( i );
                }
            }

            Console.WriteLine( num );
        }
    }
}

  2. El juego de master mind consiste en acertar un número entero que el ordenador ha escogido al azar. El jugador tiene todos los intentos que desee hasta acertar el número.

    Para esto, es necesario utilizar bucles, la clase Random y el manejo de la consola.

    Añade al código fuente más abajo la posibilidad del conteo de intentos.

    Código fuente 
    // Mastermind (c) 2021 Baltasar MIT License <baltasarq@uvigo.gal>


namespace Mastermind {
    using System;


    class Program
    {
        static void Main(string[] args)
        {
            var rnd = new Random();
            string entrada;
            int num;
            int numUsr;

            do {
                num = rnd.Next() % 100;
                Console.WriteLine( "MASTERMIND: Adivina el número (0-99)" );

                do {
                    do {
                        Console.WriteLine( "Tu intento: ");
                        entrada = Console.ReadLine().Trim();
                    } while ( !int.TryParse( entrada, out numUsr ) );

                    if( numUsr < num ) {
                        Console.WriteLine( "EL que he pensado es mayor..." );
                    }
                    else
                    if( numUsr > num ) {
                        Console.WriteLine( "EL que he pensado es menor..." );
                    }

                } while( numUsr != num );

                Console.WriteLine( "¡Acertaste!" );
                Console.WriteLine( "Jugar otra vez (S/N): " );
                entrada = Console.ReadLine().Trim().ToUpper();
            } while( entrada[ 0 ] == 'S' );
        }
    }
}

  3. Una terna pitagórica es un conjunto de tres números a, b y c, para los que se cumple que a² + b² = c². Crea un programa que compruebe, para tres valores introducidos por teclado, si se trata de una terna pitagórica o no.

    Código fuente 
    // Terna pitagórica (c) 2021 Baltasar MIT License <baltasarq@uvigo.gal>

namespace TernaPitagorica {
    using System;


    class Program {
        static void Main(string[] args)
        {
            int[] v = new int[3];

            Console.WriteLine("Terna pitagórica: v1² + v2² = v3²\n");

            for(int i = 0; i < v.Length; i++) {
                Console.Write( "Dame el valor " + ( i + 1 ) + ": " );
                v[ i ] = Convert.ToInt32( Console.ReadLine() );
            }

            if ( ( ( v[ 0 ] * v[ 0 ] ) + ( v[ 1 ] * v[ 1 ] ) ) == ( v[ 2 ] * v[ 2 ] ) ) {
                Console.WriteLine( "Es una terna pitagórica!!" );
            } else {
                Console.WriteLine( "NO es una terna pitagórica!!" );
            }
        }
    }
}

    1. Crea un programa que contenga la clase libro, y una colección de libros. El programa permitirá introducir, buscar y listar (todos los) libros. Será muy interesante el uso de la colección Dictionary para almacenar objetos de la clase libro.

      Código fuente 
                      // Biblioteca (c) Baltasar MIT License <baltasarq@uvigo.gal>


                  namespace Biblioteca {
                      using System;
                      using System.Collections.Generic;


                      public class Libro {
                          public Libro(string issn, string titulo, string autor)
                          {
                              this.ISSN = issn;
                              this.Titulo = titulo;
                              this.Autor = autor;
                          }

                          public string ISSN { get; set; }
                          public string Titulo { get; set; }
                          public string Autor { get; set; }

                          public override string ToString()
                          {
                              return String.Format( "{0}/ \"{1}\" ({2})",
                                                      this.ISSN, this.Titulo, this.Autor );
                          }
                      }

                      public class Libros {
                          public Libros()
                          {
                              this.libros = new Dictionary();
                          }

                          public void Inserta(Libro l)
                          {
                              this.libros.Add( l.ISSN, l );
                          }

                          public Libro? Busca(string issn)
                          {
                              Libro? toret = null;

                              this.libros.TryGetValue( issn, out toret );
                              return toret;
                          }

                          public Libro[] Todos {
                              get {
                                  var toret = new Libro[ this.libros.Count ];
                                  this.libros.Values.CopyTo( toret, 0 );

                                  return toret;
                              }
                          }

                          Dictionary libros;
                      }


                      class Program {
                          static string LeeConsola()
                          {
                              string? toret = Console.ReadLine();

                              return toret ?? "";
                          }

                          static void Main(string[] args)
                          {
                              string entrada = "";
                              var biblio = new Libros();

                              Console.WriteLine( "Biblioteca\n" );

                              while ( entrada != "0" ) {
                                  Console.WriteLine( "\n\n1. Inserta\n2. Busca\n3. Lista\n0. Salir\nOpc.: " );
                                  entrada = LeeConsola().Trim();

                                  switch( entrada ) {
                                      case "1":
                                          string issn = "";
                                          string titulo = "";
                                          string autor = "";

                                          Console.WriteLine( "Introduzca ISSN: " );
                                          issn = LeeConsola().Trim();

                                          Console.WriteLine( "Introduzca Titulo: " );
                                          titulo = LeeConsola().Trim();

                                          Console.WriteLine( "Introduzca Autor: " );
                                          autor = LeeConsola().Trim();

                                          biblio.Inserta( new Libro( issn, titulo, autor ) );
                                          break;
                                      case "2":
                                          Console.WriteLine( "Introduzca ISSN: " );
                                          entrada = LeeConsola().Trim();
                                          Libro? l = biblio.Busca( entrada );
                                          Console.WriteLine( l?.ToString() ?? "No encontrado." );
                                          break;
                                      case "3":
                                          Libro[] libros = biblio.Todos;

                                          foreach(Libro libro in libros) {
                                              Console.WriteLine( libro );
                                          }
                                          break;
                                      default:
                                          if ( entrada != "0" ) {
                                              Console.WriteLine( "Opc. incorrecta" );
                                          }
                                          break;
                                  }
                              }
                          }
                      }
                  }
